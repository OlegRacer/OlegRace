Обробка нахилу з автовибором осі та інверсією для альбомного режиму function handleOrientation(event) { let targetRoll = 0; if (isLandscape) { // Отримуємо кут повороту екрана (90 або -90) let orientationAngle = window.orientation || (screen.orientation && screen.orientation.angle) || 0; // Змінено логіку знаків: додано інверсію (-), щоб прибрати дзеркальне відображення if (orientationAngle < 0) { targetRoll = -event.beta; } else { targetRoll = event.beta; } } else { // В портретному режимі залишаємо стандартну вісь GAMMA targetRoll = event.gamma; } // Згладжування (фільтр), щоб цифри не стрибали від вібрацій скутера currentAngle = (0.85 * currentAngle) + (0.15 * targetRoll); let displayAngle = Math.round(currentAngle); // Виводимо чисте значення кута (завжди додатнє) angleElement.innerText = Math.abs(displayAngle); // Визначаємо напрямок нахилу if (displayAngle > 3) { directionElement.innerText = "Нахил праворуч 👉"; directionElement.style.color = "#ff5722"; } else if (displayAngle < -3) { directionElement.innerText = "👈 Нахил ліворуч"; directionElement.style.color = "#ff5722"; } else { directionElement.innerText = "Рівно 🏍️"; directionElement.style.color = "#00ffcc"; } }
